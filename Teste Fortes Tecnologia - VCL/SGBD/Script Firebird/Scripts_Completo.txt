		// TABELA PARAMENTROS  \\
CREATE TABLE PARAMETROS (
    PAR_ID             INTEGER NOT NULL,
    PAR_ALIQUOTA       NUMERIC(15,2),
    PAR_PRECOGASOLINA  NUMERIC(15,2),
    PAR_PRECODIESEL    NUMERIC(15,2)
);

//CHAVE
ALTER TABLE PARAMETROS ADD CONSTRAINT PK_PARAMETROS PRIMARY KEY (PAR_ID);

//GENERATOR
CREATE SEQUENCE GEN_PARID;

//TRIGUER AUTO_INC
CREATE OR ALTER TRIGGER TR_PARAMENTROS FOR PARAMETROS
ACTIVE BEFORE INSERT POSITION 0
as
BEGIN
  NEW.PAR_ID=GEN_ID(GEN_PARID,1);
END

    MOV_IMPOSTO      NUMERIC(15,2)
);


//CHAVE
ALTER TABLE MOVIMENTO ADD CONSTRAINT PK_MOVIMENTO PRIMARY KEY (MOV_ID);


//GENERATOR

//TRIGUER AUTO_INC
